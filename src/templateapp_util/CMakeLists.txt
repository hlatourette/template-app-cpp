set(LIBRARY_NAME templateapplib)

    PUBLIC
    INTERFACE
    PRIVATE
)

    PRIVATE
        templateapp_util.cpp
    INTERFACE
    PUBLIC
        FILE_SET HEADERS
            FILES
                templateapp_util.h
)

    PUBLIC
    INTERFACE
    PRIVATE
        .
)
